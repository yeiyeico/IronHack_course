  def screen_size
    begin
      size = TermInfo.tiocgwinsz(@io)
    rescue NotImplementedError
      size = [0,0]
    end
    if size[0] == 0
      size[0] = ENV.include?('LINES') ? ENV['LINES'].to_i : self.tigetnum("lines")
    end
    if size[1] == 0
      size[1] = ENV.include?('COLUMNS') ? ENV['COLUMNS'].to_i : self.tigetnum("cols")
    end
    size
  end
